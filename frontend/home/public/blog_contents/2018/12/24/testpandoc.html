int countPairing(bool matching[10]) {

    int first = -1;
    for (int i = 0; i < student; i++) {
        if (!matching[i]) {
            first = i;
            break;
        }
    }

    //base case
    if (first == -1) return 1;

    int ret = 0;
    for (int soul = first + 1; soul < student; soul++) {
        if (!matching[soul] && soulmate[first][soul]) {
            matching[first] = matching[soul] = true;
            ret += countPairing(matching);

        }
    }
    return ret;
}
